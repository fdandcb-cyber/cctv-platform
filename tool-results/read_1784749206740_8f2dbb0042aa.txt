    69→    69→// Camera density per 1000 sq ft by property type
    70→    70→function getCameraSuggestion(area: number, propType: PropertyType): { min: number; max: number; suggested: number } {
    71→    71→  const camerasPer1000sqft: Record<PropertyType, number> = {
    72→    72→    home: 1.2, apartment: 1.5, office: 1.8, shop: 2.0,
    73→    73→    warehouse: 0.8, factory: 1.0, restaurant: 2.5, hospital: 2.0,
    74→    74→    school: 1.5, villa: 1.0, other: 1.2
    75→    75→  };
    76→    76→  const rate = camerasPer1000sqft[propType] || 1.2;
    77→    77→  const suggested = Math.max(2, Math.ceil((area / 1000) * rate));
    78→    78→  return { min: Math.max(2, suggested - 2), max: suggested + 4, suggested };
    79→    79→}
    80→    80→
    81→    81→// ═══ RECORDER CONFIG ═══
    82→    82→// Real market channel counts:
    83→    83→//   DVR (analog): 4ch, 8ch, 16ch, 32ch   (max 32ch for DVR)
    84→    84→//   NVR (IP):     4ch, 8ch, 16ch, 32ch, 64ch, 128ch, 256ch
    85→    85→// If cameras exceed the largest available single unit, multiple units are suggested.
    86→    86→function getRecorderConfig(totalCameras: number, system: CameraSystem): { type: string; units: { channels: number; usedPorts: number }[]; summary: string; exceedsMax: boolean } {
    87→    87→  if (system === "wifi") {
    88→    88→    return { type: "None", units: [], summary: "WiFi cameras do not need a DVR/NVR. They record on MicroSD card or cloud.", exceedsMax: false };
    89→    89→  }
    90→    90→
    91→    91→  const type = system === "analog" ? "DVR" : "NVR";
    92→    92→  // DVR max is 32ch. NVR goes up to 256ch.
    93→    93→  const availableChannels = system === "analog" ? [4, 8, 16, 32] : [4, 8, 16, 32, 64, 128, 256];
    94→    94→  const maxSingleUnit = availableChannels[availableChannels.length - 1];
    95→    95→  const exceedsMax = totalCameras > maxSingleUnit;
    96→    96→
    97→    97→  const units: { channels: number; usedPorts: number }[] = [];
    98→    98→  let remaining = totalCameras;
    99→    99→
   100→   100→  while (remaining > 0) {
   101→   101→    // Pick the smallest channel count that covers remaining, or the biggest available
   102→   102→    const fitting = availableChannels.filter(ch => ch >= remaining);
   103→   103→    const chosen = fitting.length > 0 ? fitting[0] : availableChannels[availableChannels.length - 1];
   104→   104→    const used = Math.min(remaining, chosen);
   105→   105→    units.push({ channels: chosen, usedPorts: used });
   106→   106→    remaining -= used;
   107→   107→  }
   108→   108→
   109→   109→  // Build summary string
   110→   110→  if (units.length === 1) {
   111→   111→    const u = units[0];
   112→   112→    return { type, units, summary: `${type} ${u.channels}-Channel (${u.usedPorts} camera${u.usedPorts > 1 ? "s" : ""} connected)`, exceedsMax };
   113→   113→  }
   114→   114→
   115→   115→  const unitDescs = units.map(u => {
   116→   116→    if (u.usedPorts === u.channels) return `${u.channels}ch`;
   117→   117→    return `${u.channels}ch (${u.usedPorts} used)`;
   118→   118→  }).join(" + ");
   119→   119→
   120→   120→  const totalChannels = units.reduce((s, u) => s + u.channels, 0);
   121→   121→  return {
   122→   122→    type,
   123→   123→    units,
   124→   124→    summary: `${units.length}x ${type} units: ${unitDescs} = ${totalChannels} total channels for ${totalCameras} cameras`,
   125→   125→    exceedsMax
   126→   126→  };
   127→   127→}
   128→   128→
   129→   129→// ═══ POWER SUPPLY CONFIG ═══
   130→   130→// Real market PoE switch ports: 4, 8, 16, 24  (32-port and 48-port PoE switches do NOT exist)
   131→   131→// SMPS channels for analog: 4ch, 8ch, 16ch
   132→   132→function getPowerConfig(totalCameras: number, system: CameraSystem, hasAbove2mp: boolean): { units: { type: string; ports: number; usedPorts: number; variant: "standard" | "giga" }[]; summary: string } {
   133→   133→  if (system === "wifi") {
   134→   134→    return { units: [], summary: "WiFi cameras use included 12V adapters. No separate power supply needed." };
   135→   135→  }
   136→   136→
   137→   137→  if (system === "analog") {
   138→   138→    // SMPS: 4ch, 8ch, 16ch
   139→   139→    const smpsOptions = [4, 8, 16];
   140→   140→    const units: { type: string; ports: number; usedPorts: number; variant: "standard" | "giga" }[] = [];
   141→   141→    let remaining = totalCameras;
   142→   142→
   143→   143→    while (remaining > 0) {
   144→   144→      const fitting = smpsOptions.filter(ch => ch >= remaining);
   145→   145→      const chosen = fitting.length > 0 ? fitting[0] : smpsOptions[smpsOptions.length - 1];
   146→   146→      const used = Math.min(remaining, chosen);
   147→   147→      units.push({ type: "SMPS Power Supply", ports: chosen, usedPorts: used, variant: "standard" });
   148→   148→      remaining -= used;
   149→   149→    }
   150→   150→
   151→   151→    const unitDescs = units.map(u => {
   152→   152→      if (u.usedPorts === u.ports) return `${u.ports}-Channel SMPS`;
   153→   153→      return `${u.ports}-Channel SMPS (${u.usedPorts} used)`;
   154→   154→    }).join(" + ");
   155→   155→
   156→   156→    return { units, summary: `${units.length}x SMPS units: ${unitDescs} (12V DC)` };
   157→   157→  }
   158→   158→
   159→   159→  // IP system - PoE Switch
   160→   160→  // Real market: 4-port, 8-port, 16-port, 24-port PoE switches (max 24)
   161→   161→  // Gigabit needed for >2MP cameras (higher bandwidth)
   162→   162→  const poeOptions = [4, 8, 16, 24];
   163→   163→  const variant: "standard" | "giga" = hasAbove2mp ? "giga" : "standard";
   164→   164→  const units: { type: string; ports: number; usedPorts: number; variant: "standard" | "giga" }[] = [];
   165→   165→  let remaining = totalCameras;
   166→   166→
   167→   167→  while (remaining > 0) {
   168→   168→    const fitting = poeOptions.filter(ch => ch >= remaining);
   169→   169→    const chosen = fitting.length > 0 ? fitting[0] : poeOptions[poeOptions.length - 1];
   170→   170→    const used = Math.min(remaining, chosen);
   171→   171→    units.push({ type: "PoE Switch", ports: chosen, usedPorts: used, variant });
   172→   172→    remaining -= used;
   173→   173→  }
   174→   174→
   175→   175→  const variantLabel = variant === "giga" ? "Gigabit" : "Fast Ethernet";
   176→   176→  const unitDescs = units.map(u => {
   177→   177→    if (u.usedPorts === u.ports) return `${u.ports}-Port ${variantLabel}`;
   178→   178→    return `${u.ports}-Port (${u.usedPorts} used)`;
   179→   179→  }).join(" + ");
   180→   180→
   181→   181→  const totalPorts = units.reduce((s, u) => s + u.ports, 0);
   182→   182→  return { units, summary: `${units.length}x ${variantLabel} PoE switches: ${unitDescs} = ${totalPorts} total ports for ${totalCameras} cameras` };
   183→   183→}
   184→   184→
   185→   185→// ═══ HDD SIZE CALCULATION ═══
   186→   186→// Real-world H.265+ bitrates (Mbps) at 15fps continuous recording:
   187→   187→//   2MP  (1080p): 2 Mbps   → ~21.6 GB/day/camera
   188→   188→//   3MP:          3 Mbps   → ~32.4 GB/day/camera
   189→   189→//   4MP  (2K):    4 Mbps   → ~43.2 GB/day/camera
   190→   190→//   5MP:          6 Mbps   → ~64.8 GB/day/camera
   191→   191→//   8MP  (4K):    8 Mbps   → ~86.4 GB/day/camera
   192→   192→//   12MP:         16 Mbps  → ~172.8 GB/day/camera
   193→   193→// With H.265+ motion recording (~70% activity factor), multiply by 0.7
   194→   194→// Formula per camera: GB/day = (bitrate_Mbps / 8) * 3600 * 24 * 0.7
   195→   195→// Total GB = Sum(camera_qty_i * GB_per_day_i * retention_days)
   196→   196→
   197→   197→const BITRATE_GB_PER_DAY: Record<string, number> = {
   198→   198→  "1MP": 7.6,     // (1/8)*86400*0.7
   199→   199→  "2MP": 15.1,    // (2/8)*86400*0.7 = 15120 MB = 14.8 GB → rounded
   200→   200→  "3MP": 22.7,
   201→   201→  "4MP": 30.2,
   202→   202→  "5MP": 45.4,
   203→   203→  "8MP": 60.5,
   204→   204→  "12MP": 120.9
   205→   205→};
   206→   206→
   207→   207→const HDD_SIZES_GB = [500, 1000, 2000, 3000, 4000, 6000, 8000, 10000, 12000, 16000];
   208→   208→
   209→   209→function calculateHdd(cameraSelections: CameraSelection[], retentionDays: number): { size: string; breakdown: string } {
   210→   210→  if (cameraSelections.length === 0 || retentionDays <= 0) return { size: "", breakdown: "" };
   211→   211→
   212→   212→  let totalGB = 0;
   213→   213→  const details: string[] = [];
   214→   214→
   215→   215→  for (const cam of cameraSelections) {
   216→   216→    const gbPerDay = BITRATE_GB_PER_DAY[cam.mp] || 30.2;
   217→   217→    const camTotal = gbPerDay * cam.qty * retentionDays;
   218→   218→    totalGB += camTotal;
   219→   219→    details.push(`${cam.qty}x ${cam.mp} ${cam.form}: ${Math.round(camTotal / 1024 * 10) / 10} TB`);
   220→   220→  }
   221→   221→
   222→   222→  // Find smallest standard HDD that fits
   223→   223→  let hddLabel = "";
   224→   224→  for (const size of HDD_SIZES_GB) {
   225→   225→    if (size >= totalGB) {
   226→   226→      hddLabel = size >= 1000 ? `${size / 1000} TB` : `${size} GB`;
   227→   227→      break;
   228→   228→    }
   229→   229→  }
   230→   230→  if (!hddLabel) {
   231→   231→    const tb = Math.ceil(totalGB / 1000);
   232→   232→    hddLabel = `${tb}+ TB (Multiple HDDs needed)`;
   233→   233→  }
   234→   234→
   235→   235→  // Build breakdown
   236→   236→  const totalTB = (totalGB / 1024).toFixed(1);
   237→   237→  const breakdown = `Total: ~${totalTB} TB for ${retentionDays} days retention`;
   238→   238→
   239→   239→  return { size: hddLabel, breakdown };
   240→   240→}

   242→   242→function getTotalCableMeters(cameraQty: number, metersPerCam: number): number {
   243→   243→  return cameraQty * metersPerCam;
   244→   244→}

   246→   246→function getCableRolls(totalMeters: number, rollLength: number): number {

   248→   248→}

   289→   289→
   290→   290→export function CctvBuilder() {
   291→   291→  const store = useBuilderStore();
   292→   292→  const router = useRouter();
   293→   293→
   294→   294→  // ─── Derived State ───
   295→   295→  const area = parseFloat(store.areaSqft) || 0;
   296→   296→  const hasPropertyType = store.propertyType !== "";
   297→   297→  const hasArea = area > 0;
   298→   298→  const hasSystem = store.cameraSystem !== "";
   299→   299→  const hasTech = store.cameraTechs.length > 0;
   300→   300→
   301→   301→  const cameraSuggestion = useMemo(() => {
   302→   302→    if (!hasPropertyType || !hasArea) return null;
   303→   303→    return getCameraSuggestion(area, store.propertyType as PropertyType);
   304→   304→  }, [hasPropertyType, hasArea, area, store.propertyType]);
   305→   305→
   306→   306→  const totalCameras = useMemo(() => store.cameraSelections.reduce((s, c) => s + c.qty, 0), [store.cameraSelections]);
   307→   307→  const hasAbove2mp = useMemo(() => store.cameraSelections.some(c => c.mp !== "2MP"), [store.cameraSelections]);
   308→   308→
   309→   309→  // ─── Auto-calculated: Recorder Config ───
   310→   310→  const recorderConfig = useMemo(() => {
   311→   311→    if (totalCameras === 0) return null;
   312→   312→    return getRecorderConfig(totalCameras, store.cameraSystem as CameraSystem);
   313→   313→  }, [totalCameras, store.cameraSystem]);
   314→   314→
   315→   315→  // ─── Auto-calculated: Power Config ───
   316→   316→  const powerConfig = useMemo(() => {
   317→   317→    if (totalCameras === 0) return null;
   318→   318→    return getPowerConfig(totalCameras, store.cameraSystem as CameraSystem, hasAbove2mp);
   319→   319→  }, [totalCameras, store.cameraSystem, hasAbove2mp]);
   320→   320→
   321→   321→  // ─── Auto-calculated: HDD Size ───
   322→   322→  const hddCalc = useMemo(() => {
   323→   323→    if (totalCameras === 0 || store.cameraSystem === "wifi") return { size: "", breakdown: "" };
   324→   324→    return calculateHdd(store.cameraSelections, store.retentionDays);
   325→   325→  }, [totalCameras, store.cameraSelections, store.retentionDays, store.cameraSystem]);
   326→   326→
   327→   327→  // ─── Auto-calculated: Cable ───
   328→   328→  const totalCableMeters = useMemo(() => getTotalCableMeters(totalCameras, store.cableLengthPerCamera), [totalCameras, store.cableLengthPerCamera]);
   329→   329→  const cableRolls90 = useMemo(() => getCableRolls(totalCableMeters, 90), [totalCableMeters]);
   330→   330→  const cableRolls180 = useMemo(() => getCableRolls(totalCableMeters, 180), [totalCableMeters]);
   331→   331→  const cableRolls100 = useMemo(() => getCableRolls(totalCableMeters, 100), [totalCableMeters]);
   332→   332→  const cableRolls305 = useMemo(() => getCableRolls(totalCableMeters, 305), [totalCableMeters]);
   333→   333→
   334→   334→  // ─── Sync auto-calculated values to store ───
   335→   335→  useEffect(() => {
   336→   336→    if (recorderConfig) {
   337→   337→      store.setRecorderUnits(recorderConfig.units.map(u => ({ ...u, type: recorderConfig.type })));
   338→   338→    } else {
   339→   339→      store.setRecorderUnits([]);
   340→   340→    }
   341→   341→  }, [recorderConfig, store.setRecorderUnits]);
