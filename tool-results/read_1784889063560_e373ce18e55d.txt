   118→   118→function getCameraSuggestion(area: number, propType: PropertyType): { min: number; max: number; suggested: number } {
   119→   119→  const camerasPer1000sqft: Record<PropertyType, number> = {
   120→   120→    home: 1.2, apartment: 1.5, office: 1.8, shop: 2.0,
   121→   121→    warehouse: 0.8, factory: 1.0, restaurant: 2.5, hospital: 2.0,
   122→   122→    school: 1.5, villa: 1.0, other: 1.2
   123→   123→  };
   124→   124→  const rate = camerasPer1000sqft[propType] || 1.2;
   125→   125→  const suggested = Math.max(2, Math.ceil((area / 1000) * rate));
   126→   126→  return { min: Math.max(2, suggested - 2), max: suggested + 4, suggested };
   127→   127→}

   129→   129→function getRecorderConfig(totalCameras: number, system: CameraSystem): { type: string; units: { channels: number; usedPorts: number }[]; summary: string; exceedsMax: boolean } {
   130→   130→  if (system === "wifi") {
   131→   131→    return { type: "None", units: [], summary: "WiFi cameras do not need a DVR/NVR. They record on MicroSD card or cloud.", exceedsMax: false };
   132→   132→  }
   133→   133→  const type = system === "analog" ? "DVR" : "NVR";
   134→   134→  const availableChannels = system === "analog" ? [4, 8, 16, 32] : [4, 8, 16, 32, 64, 128, 256];
   135→   135→  const maxSingleUnit = availableChannels[availableChannels.length - 1];
   136→   136→  const exceedsMax = totalCameras > maxSingleUnit;
   137→   137→  const units: { channels: number; usedPorts: number }[] = [];
   138→   138→  let remaining = totalCameras;
   139→   139→  while (remaining > 0) {
   140→   140→    const fitting = availableChannels.filter(ch => ch >= remaining);
   141→   141→    const chosen = fitting.length > 0 ? fitting[0] : availableChannels[availableChannels.length - 1];
   142→   142→    const used = Math.min(remaining, chosen);
   143→   143→    units.push({ channels: chosen, usedPorts: used });
   144→   144→    remaining -= used;
   145→   145→  }
   146→   146→  if (units.length === 1) {
   147→   147→    const u = units[0];
   148→   148→    return { type, units, summary: `${type} ${u.channels}-Channel (${u.usedPorts} camera${u.usedPorts > 1 ? 's' : ''} connected)`, exceedsMax };
   149→   149→  }
   150→   150→  const unitDescs = units.map(u => u.usedPorts === u.channels ? `${u.channels}ch` : `${u.channels}ch (${u.usedPorts} used)`).join(" + ");
   151→   151→  const totalChannels = units.reduce((s, u) => s + u.channels, 0);
   152→   152→  return { type, units, summary: `${units.length}x ${type} units: ${unitDescs} = ${totalChannels} total channels for ${totalCameras} cameras`, exceedsMax };
   153→   153→}

   155→   155→function getPowerConfig(totalCameras: number, system: CameraSystem, hasAbove2mp: boolean): { units: { type: string; ports: number; usedPorts: number; variant: "standard" | "giga" }[]; summary: string } {
   156→   156→  if (system === "wifi") return { units: [], summary: "WiFi cameras use included 12V adapters. No separate power supply needed." };
   157→   157→  if (system === "analog") {
   158→   158→    const smpsOptions = [4, 8, 16];
   159→   159→    const units: { type: string; ports: number; usedPorts: number; variant: "standard" | "giga" }[] = [];
   160→   160→    let remaining = totalCameras;
   161→   161→    while (remaining > 0) {
   162→   162→      const fitting = smpsOptions.filter(ch => ch >= remaining);
   163→   163→      const chosen = fitting.length > 0 ? fitting[0] : smpsOptions[smpsOptions.length - 1];
   164→   164→      const used = Math.min(remaining, chosen);
   165→   165→      units.push({ type: "SMPS Power Supply", ports: chosen, usedPorts: used, variant: "standard" });
   166→   166→      remaining -= used;
   167→   167→    }
   168→   168→    const unitDescs = units.map(u => u.usedPorts === u.ports ? `${u.ports}-Channel SMPS` : `${u.ports}-Channel SMPS (${u.usedPorts} used)`).join(" + ");
   169→   169→    return { units, summary: `${units.length}x SMPS units: ${unitDescs} (12V DC)` };
   170→   170→  }
   171→   171→  const poeOptions = [4, 8, 16, 24];
   172→   172→  const variant: "standard" | "giga" = hasAbove2mp ? "giga" : "standard";
   173→   173→  const units: { type: string; ports: number; usedPorts: number; variant: "standard" | "giga" }[] = [];
   174→   174→  let remaining = totalCameras;
   175→   175→  while (remaining > 0) {
   176→   176→    const fitting = poeOptions.filter(ch => ch >= remaining);
   177→   177→    const chosen = fitting.length > 0 ? fitting[0] : poeOptions[poeOptions.length - 1];
   178→   178→    const used = Math.min(remaining, chosen);
   179→   179→    units.push({ type: "PoE Switch", ports: chosen, usedPorts: used, variant });
   180→   180→    remaining -= used;
   181→   181→  }
   182→   182→  const variantLabel = variant === "giga" ? "Gigabit" : "Fast Ethernet";
   183→   183→  const unitDescs = units.map(u => u.usedPorts === u.ports ? `${u.ports}-Port ${variantLabel}` : `${u.ports}-Port (${u.usedPorts} used)`).join(" + ");
   184→   184→  return { units, summary: `${units.length}x ${variantLabel} PoE switches: ${unitDescs}` };
   185→   185→}

   187→   187→const BITRATE_GB_PER_DAY: Record<string, number> = { "1MP": 7.6, "2MP": 15.1, "3MP": 22.7, "4MP": 30.2, "5MP": 45.4, "8MP": 60.5, "12MP": 120.9 };
   188→   188→const HDD_SIZES_GB = [500, 1000, 2000, 3000, 4000, 6000, 8000, 10000, 12000, 16000];

   190→   190→function calculateHdd(cameraSelections: CameraSelection[], retentionDays: number): { size: string; breakdown: string } {
   191→   191→  if (cameraSelections.length === 0 || retentionDays <= 0) return { size: "", breakdown: "" };
   192→   192→  let totalGB = 0;
   193→   193→  for (const cam of cameraSelections) {
   194→   194→    const gbPerDay = BITRATE_GB_PER_DAY[cam.mp] || 30.2;
   195→   195→    totalGB += gbPerDay * cam.qty * retentionDays;
   196→   196→  }
   197→   197→  let hddLabel = "";
   198→   198→  for (const size of HDD_SIZES_GB) {
   199→   199→    if (size >= totalGB) { hddLabel = size >= 1000 ? `${size / 1000} TB` : `${size} GB`; break; }
   200→   200→  }
   201→   201→  if (!hddLabel) hddLabel = `${Math.ceil(totalGB / 1000)}+ TB (Multiple HDDs needed)`;
   202→   202→  return { size: hddLabel, breakdown: `Total: ~${(totalGB / 1024).toFixed(1)} TB for ${retentionDays} days retention` };
   203→   203→}
