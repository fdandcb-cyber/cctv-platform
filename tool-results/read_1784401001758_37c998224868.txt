   134→   134→// SAVED QUOTES MANAGEMENT
   135→   135→// ═══════════════════════════════════════════════════════════════
   136→   136→
   137→   137→interface SavedQuote {
   138→   138→  id: string;
   139→   139→  name: string;
   140→   140→  date: string;
   141→   141→  data: string; // JSON string of builder state
   142→   142→  summary: string;
   143→   143→}
   144→   144→
   145→   145→function QuotesTab() {
   146→   146→  const [quotes, setQuotes] = useState<SavedQuote[]>([]);
   147→   147→  const [loadedQuote, setLoadedQuote] = useState<string | null>(null);
   148→   148→
   149→   149→  useEffect(() => {
   150→   150→    try {
   151→   151→      const stored = localStorage.getItem("cctv_saved_quotes");
   152→   152→      if (stored) setQuotes(JSON.parse(stored));
   153→   153→    } catch { /* empty */ }
   154→   154→  }, []);
   155→   155→
   156→   156→  const saveQuote = () => {
   157→   157→    const state = useBuilderStore.getState();
   158→   158→    const totalCams = state.cameraSelections.reduce((s, c) => s + c.qty, 0);
   159→   159→    if (totalCams === 0) {
   160→   160→      toast.error("No cameras configured. Build a setup first!");
   161→   161→      return;
   162→   162→    }
   163→   163→    const name = prompt("Quote name (e.g., Customer Name - Site):") || "Untitled Quote";
   164→   164→    const quote: SavedQuote = {
   165→   165→      id: Date.now().toString(),
   166→   166→      name,
   167→   167→      date: new Date().toLocaleDateString("en-IN"),
   168→   168→      data: JSON.stringify({
   169→   169→        propertyType: state.propertyType,
   170→   170→        areaSqft: state.areaSqft,
   171→   171→        cameraSystem: state.cameraSystem,
   172→   172→        cameraTechs: state.cameraTechs,
   173→   173→        cameraSelections: state.cameraSelections,
   174→   174→        recorderUnits: state.recorderUnits,
   175→   175→        powerUnits: state.powerUnits,
   176→   176→        retentionDays: state.retentionDays,
   177→   177→        hddSuggestion: state.hddSuggestion,
   178→   178→        cableLengthPerCamera: state.cableLengthPerCamera,
   179→   179→        cableSelection: state.cableSelection,
   180→   180→        junctionBox4x4: state.junctionBox4x4,
   181→   181→        junctionBox5x5: state.junctionBox5x5,
   182→   182→        dcConnector: state.dcConnector,
   183→   183→        bncConnector: state.bncConnector,
   184→   184→        rj45Connector: state.rj45Connector,
   185→   185→        networkingRack: state.networkingRack,
   186→   186→        monitor: state.monitor,
   187→   187→        monitorSize: state.monitorSize,
   188→   188→        extensionBoard: state.extensionBoard,
   189→   189→        wirelessMouse: state.wirelessMouse,
   190→   190→        wirelessKeyboard: state.wirelessKeyboard,
   191→   191→        ups: state.ups,
   192→   192→      }),
   193→   193→      summary: `${totalCams} cameras | ${state.cameraSystem.toUpperCase()} | ${state.areaSqft} sq ft`,
   194→   194→    };
   195→   195→    const updated = [quote, ...quotes];
   196→   196→    setQuotes(updated);
   197→   197→    localStorage.setItem("cctv_saved_quotes", JSON.stringify(updated));
   198→   198→    toast.success("Quote saved!");
   199→   199→  };
   200→   200→
   201→   201→  const deleteQuote = (id: string) => {
   202→   202→    if (!confirm("Delete this saved quote?")) return;
   203→   203→    const updated = quotes.filter(q => q.id !== id);
   204→   204→    setQuotes(updated);
   205→   205→    localStorage.setItem("cctv_saved_quotes", JSON.stringify(updated));
   206→   206→    toast.success("Quote deleted");
   207→   207→  };
   208→   208→
   209→   209→  const loadQuote = (quote: SavedQuote) => {
   210→   210→    try {
   211→   211→      const data = JSON.parse(quote.data);
   212→   212→      const store = useBuilderStore.getState();
   213→   213→      Object.keys(data).forEach((key) => {
   214→   214→        const actionKey = "set" + key.charAt(0).toUpperCase() + key.slice(1) as keyof typeof store;
   215→   215→        if (typeof store[actionKey] === "function") {
   216→   216→          (store[actionKey] as (v: unknown) => void)(data[key]);
   217→   217→        }
   218→   218→      });
   219→   219→      setLoadedQuote(quote.id);
   220→   220→      toast.success("Quote loaded! Go to Builder tab to view.");
   221→   221→      setTimeout(() => setLoadedQuote(null), 3000);
   222→   222→    } catch {
   223→   223→      toast.error("Failed to load quote");
   224→   224→    }
   225→   225→  };
   226→   226→
   227→   227→  const copyQuoteText = (quote: SavedQuote) => {
   228→   228→    try {
   229→   229→      const data = JSON.parse(quote.data);
   230→   230→      const lines: string[] = [];
   231→   231→      lines.push("=== CCTV QUOTE ===");
   232→   232→      lines.push("Name: " + quote.name);
   233→   233→      lines.push("Date: " + quote.date);
   234→   234→      lines.push("Property: " + data.propertyType);
   235→   235→      lines.push("Area: " + data.areaSqft + " sq ft");
   236→   236→      lines.push("System: " + (data.cameraSystem || "").toUpperCase());
   237→   237→      lines.push("");
   238→   238→      if (data.cameraSelections) {
   239→   239→        lines.push("--- CAMERAS ---");
   240→   240→        let total = 0;
   241→   241→        for (const cam of data.cameraSelections) {
   242→   242→          lines.push(cam.qty + "x " + cam.mp + " " + cam.form);
   243→   243→          total += cam.qty;
   244→   244→        }
   245→   245→        lines.push("Total: " + total + " cameras");
   246→   246→      }
   247→   247→      if (data.recorderUnits && data.recorderUnits.length > 0) {
   248→   248→        lines.push("");
   249→   249→        lines.push("--- RECORDER ---");
   250→   250→        for (const u of data.recorderUnits) {
   251→   251→          lines.push(u.channels + "-Channel " + u.type + " (" + u.usedPorts + " used)");
   252→   252→        }
   253→   253→      }
   254→   254→      if (data.powerUnits && data.powerUnits.length > 0) {
   255→   255→        lines.push("");
   256→   256→        lines.push("--- POWER ---");
   257→   257→        for (const u of data.powerUnits) {
   258→   258→          lines.push(u.ports + "-" + (u.type.includes("SMPS") ? "Channel SMPS" : "Port PoE") + " (" + u.usedPorts + " used)");
   259→   259→        }
   260→   260→      }
   261→   261→      if (data.hddSuggestion) {
   262→   262→        lines.push("");
   263→   263→        lines.push("--- STORAGE ---");
   264→   264→        lines.push("HDD: " + data.hddSuggestion);
   265→   265→        lines.push("Retention: " + data.retentionDays + " days");
   266→   266→      }
   267→   267→      lines.push("");
   268→   268→      lines.push("--- ACCESSORIES ---");
   269→   269→      if (data.junctionBox4x4 > 0) lines.push("Junction Box 4x4: " + data.junctionBox4x4);
   270→   270→      if (data.junctionBox5x5 > 0) lines.push("Junction Box 5x5: " + data.junctionBox5x5);
   271→   271→      if (data.dcConnector > 0) lines.push("DC Connector: " + data.dcConnector);
   272→   272→      if (data.bncConnector > 0) lines.push("BNC Connector: " + data.bncConnector);
   273→   273→      if (data.rj45Connector > 0) lines.push("RJ45 Connector: " + data.rj45Connector);
   274→   274→      if (data.networkingRack > 0) lines.push("Networking Rack: " + data.networkingRack);
   275→   275→      if (data.monitor > 0) lines.push("Monitor: " + data.monitor + "x " + data.monitorSize);
   276→   276→      if (data.extensionBoard > 0) lines.push("Extension Board: " + data.extensionBoard);
   277→   277→      if (data.wirelessMouse > 0) lines.push("Wireless Mouse: " + data.wirelessMouse);
   278→   278→      if (data.ups > 0) lines.push("UPS: " + data.ups);
   279→   279→
   280→   280→      navigator.clipboard.writeText(lines.join("\n")).then(() => toast.success("Quote copied!")).catch(() => toast.error("Copy failed"));
   281→   281→    } catch { toast.error("Failed to read quote data"); }
   282→   282→  };
   283→   283→
   284→   284→  return (
   285→   285→    <div className="space-y-4">
   286→   286→      <div className="flex items-center justify-between">
   287→   287→        <div>
   288→   288→          <h3 className="font-semibold">Saved Quotes</h3>
   289→   289→          <p className="text-xs text-muted-foreground">Save and manage CCTV configuration quotes. Quotes are stored in your browser.</p>
   290→   290→        </div>
   291→   291→        <Button size="sm" className="gap-1.5" onClick={saveQuote}>
   292→   292→          <Download className="h-3.5 w-3.5" /> Save Current Config
   293→   293→        </Button>
   294→   294→      </div>
   295→   295→
   296→   296→      {quotes.length === 0 ? (
   297→   297→        <div className="text-center py-12 bg-muted/30 rounded-xl border border-dashed">
   298→   298→          <FileText className="h-10 w-10 mx-auto text-muted-foreground mb-2" />
   299→   299→          <p className="text-sm text-muted-foreground">No saved quotes yet. Configure a setup in the Builder tab, then save it here.</p>
   300→   300→        </div>
   301→   301→      ) : (
   302→   302→        <div className="space-y-2">
   303→   303→          {quotes.map((q) => (
   304→   304→            <div key={q.id} className={cn(
   305→   305→              "flex items-center gap-4 p-4 rounded-xl border-2 transition-all",
   306→   306→              loadedQuote === q.id ? "border-emerald-400 bg-emerald-50" : "border-muted hover:border-emerald-300"
   307→   307→            )}>
   308→   308→              <div className="bg-emerald-100 text-emerald-700 p-2.5 rounded-lg shrink-0">
   309→   309→                <FileText className="h-5 w-5" />
   310→   310→              </div>
   311→   311→              <div className="flex-1 min-w-0">
   312→   312→                <p className="font-semibold text-sm truncate">{q.name}</p>
   313→   313→                <p className="text-xs text-muted-foreground">{q.summary}</p>
   314→   314→                <p className="text-[11px] text-muted-foreground mt-0.5 flex items-center gap-1">
   315→   315→                  <Clock className="h-3 w-3" /> {q.date}
   316→   316→                </p>
   317→   317→              </div>
   318→   318→              <div className="flex items-center gap-1.5 shrink-0">
   319→   319→                <Button size="sm" variant="outline" className="gap-1 text-xs h-8" onClick={() => loadQuote(q)}>
   320→   320→                  <Download className="h-3 w-3" /> Load
   321→   321→                </Button>
   322→   322→                <Button size="sm" variant="outline" className="gap-1 text-xs h-8" onClick={() => copyQuoteText(q)}>
   323→   323→                  <ClipboardCopy className="h-3 w-3" /> Copy
   324→   324→                </Button>
   325→   325→                <Button size="sm" variant="ghost" className="text-red-400 hover:text-red-600 h-8 w-8 p-0" onClick={() => deleteQuote(q.id)}>
   326→   326→                  <Trash className="h-3.5 w-3.5" />
   327→   327→                </Button>
   328→   328→              </div>
   329→   329→            </div>
   330→   330→          ))}
   331→   331→        </div>
   332→   332→      )}
   333→   333→    </div>
   334→   334→  );
   335→   335→}
